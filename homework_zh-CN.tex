\begin{problem}
    \textbf{学习目标概览。}本题展示 `homework` 文档类的核心要素：
    \begin{itemize}
        \item 自动页眉页脚与带框题面（`problemstyle=boxed`）。
        \item 紧凑列表（`tightlists`）与自适应行距、页边距配置宏。
        \item `mathdelims`/`statmacros` 提供的常用成对定界符和概率记号。
    \end{itemize}

    假设随机变量 $X$ 与 $Y$ 满足
    $\Expect{X}=0$、$\Var{X}=1$、$\Cov(X,Y)=\rho$。请完成下列子任务：
    \begin{enumerate}
        \item 证明 $\abs{\Cov(X,Y)} \leq \norm{Y}_2$。
        \item 写出 $\Prob{\abs{Y}\leq 1}$ 的上下界，并化简 $\indicator{Y>0}$ 的期望。
        \item 给出 $\BiasOf{\hat\theta}$ 的计算式，并判断其是否为 0。
    \end{enumerate}

    \solution

    \begin{enumerate}
        \item 由柯西-舒瓦兹不等式有
            $\abs{\Cov(X,Y)}=\abs{\Expect{XY}}\leq\norm{X}_2\norm{Y}_2=\norm{Y}_2$。
        \item 注意 $\Prob{\abs{Y}\leq 1}=1-\Prob{\abs{Y}>1}$，结合马尔可夫不等式可得
            $\Prob{\abs{Y}>1}\leq\Expect{\abs{Y}}$，从而得到上下界。又
            $\Expect{\indicator{Y>0}}=\Prob{Y>0}$。
        \item 设估计量 $\hat\theta$，则
            $\BiasOf{\hat\theta}=\Expect{\hat\theta}-\theta$。若 $\Expect{\hat\theta}=\theta$，偏差为 0。
    \end{enumerate}

    \begin{note}[延伸阅读]
        若需要进一步探究，可在此处补充教材章节或参考资料。
    \end{note}
\end{problem}

\begin{problem}[矩估计与极大似然]
    下列两个自变量的样本 $\{X_i\}_{i=1}^n$ 与 $\{Y_i\}_{i=1}^n$ 来自参数为 $\theta$ 的同一族分布。
    请基于课堂方法完成题目。

    \begin{subproblem}
        写出利用样本矩构造的估计量，并说明是否无偏。
    \end{subproblem}
    \begin{subproblem}
        给出极大似然方程，讨论 Newton 法迭代初值的选取策略。
    \end{subproblem}
    \begin{subproblem*}[补充讨论]
        比较两种估计的方差，并说明何时选择其中之一更为稳健。
    \end{subproblem*}

    \solution

    \begin{enumerate}
        \item 矩估计令样本矩等于理论矩即可；若模型线性，则可无偏。
        \item 极大似然方程可以写成 $\partial\ell(\theta)/\partial\theta=0$，常用 $\hat\theta_{\mathrm{MoM}}$ 作为初值。
        \item 若数据存在离群点，优先选择方差更小的估计或结合截尾策略。
    \end{enumerate}
\end{problem}

\begin{problem*}
    本题演示题目标题的手动覆写与 \verb|problem*| 不累加编号的效果。
    请根据以下要求完成：
    \begin{enumerate}
        \item 证明若 $f\colon \mathbb{R}\to\mathbb{R}$ 可导且 $f(0)=0$，则存在 $\xi\in(0,1)$ 使得
            $f(1)=f'(\xi)$。
        \item 使用 \verb|proof| 环境给出严格论证，并在结尾自动生成 $\blacksquare$。
    \end{enumerate}

    \solution

    \begin{proof}
        令 $g(x)=f(x)-f(0)-f'(0)x$，则 $g(0)=g(1)=0$。根据拉格朗日中值定理，存在
        $\xi\in(0,1)$ 使得 $g'(\xi)=0$，进而得到 $f(1)=f'(\xi)$。
    \end{proof}
